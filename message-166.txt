-- Função de movimento do Speed
local function getMovementDirection()
    local character = player.Character
    if not character then return Vector3.zero end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    return humanoid and humanoid.MoveDirection or Vector3.zero
end

local function startSpeedBoost2()
    if speedConnection2 then return end
    
    speedConnection2 = RunService.Heartbeat:Connect(function()
        if not speedBoostEnabled then return end
        
        pcall(function()
            local character = player.Character
            if not character then return end
            
            local rootPart = character:FindFirstChild("HumanoidRootPart")
            if not rootPart then return end
            
            local moveDir = getMovementDirection()
            
            if moveDir.Magnitude > 0.1 then
                rootPart.AssemblyLinearVelocity = Vector3.new(
                    moveDir.X * boostSpeed,
                    rootPart.AssemblyLinearVelocity.Y,
                    moveDir.Z * boostSpeed
                )
            end
        end)
    end)
end

local function stopSpeedBoost2()
    if speedConnection2 then
        speedConnection2:Disconnect()
        speedConnection2 = nil
    end
end

            speedBoostEnabled = false
            if speedButton then
                speedButton.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
            end
            stopSpeedBoost2()
        end
    end

-- FUNÇÃO PARA PEGAR JOGADOR MAIS PRÓXIMO
local function getNearestPlayer()
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local root = char.HumanoidRootPart
    local nearest, distance = nil, math.huge
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local mag = (root.Position - p.Character.HumanoidRootPart.Position).Magnitude
            if mag < distance then
                distance = mag
                nearest = p
            end
        end
    end
    return nearest
end

            task.wait(0.1)
        end
        root.CFrame = sequence[1]
        task.wait(0.1)
        root.CFrame = sequence[2]
    end
end

		hrp.AssemblyAngularVelocity = Vector3.zero

	end

local function RestoreMotors(character)
	for _, v in ipairs(character:GetDescendants()) do
		if v:IsA("Motor6D") then
			v.Enabled = true
		elseif v:IsA("Constraint") then
			v.Enabled = false
		end
	end
end
local function InitAntiRagdoll(character)
	local hum = character:WaitForChild("Humanoid", 10)
	if not hum then return end
	for state in pairs(BlockedStates) do
		hum:SetStateEnabled(state, false)
	end
	hum.StateChanged:Connect(function(_, new)
		if ENABLE_ANTI_RAGDOLL and BlockedStates[new] then
			ForceNormal(character)
			RestoreMotors(character)
		end
	end)
	RunService.Stepped:Connect(function()
		if not ENABLE_ANTI_RAGDOLL then
			Release(character)
			return
		end
		if BlockedStates[hum:GetState()] then
			ForceNormal(character)
		else
			Release(character)
		end
		hum.Health = hum.MaxHealth
	end)
end
local function KillRemote(remote)
	if not getconnections or not remote:IsA("RemoteEvent") then return end
	if DisabledRemotes[remote] then return end
	local name = remote.Name:lower()
	for _, key in ipairs(RemoteKeywords) do
		if name:find(key) then
			DisabledRemotes[remote] = {}
			for _, c in ipairs(getconnections(remote.OnClientEvent)) do
				if c.Disable then
					c:Disable()
					table.insert(DisabledRemotes[remote], c)
				end
			end
			break
		end
	end
end

-- Função para criar ESP em um jogador
local function createESP(plr)
    if plr == player then return end
    if not plr.Character then return end

    -- Aguarda o Character carregar completamente
    local char = plr.Character
    local head = char:WaitForChild("Head", 5)
    if not head then return end
    
    -- Remove ESP antigo se existir
    local oldESP = char:FindFirstChild("ESP_Name")
    if oldESP then
        oldESP:Destroy()
    end

    local billboard = Instance.new("BillboardGui")
    billboard.Name = "ESP_Name"
    billboard.Adornee = head
    billboard.Size = UDim2.new(0, 200, 0, 40)
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.AlwaysOnTop = true
    billboard.Parent = char

    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.Text = plr.DisplayName or plr.Name
    label.TextColor3 = Color3.fromRGB(255,255,255)
    label.Font = Enum.Font.GothamBold
    label.TextScaled = false
    label.TextSize = 10
    label.TextStrokeTransparency = 0
    label.TextStrokeColor3 = Color3.fromRGB(160,0,255)
    label.Parent = billboard
end

-- Função para remover ESP
local function removeESP(plr)
    if not plr.Character then return end
    local nameGui = plr.Character:FindFirstChild("ESP_Name")
    if nameGui then 
        nameGui:Destroy() 
    end
end

-- Função para configurar ESP em um jogador
local function setupPlayerESP(plr)
    if plr == player then return end
    
    -- Remove conexões antigas do jogador
    if espConnections[plr] then
        for _, conn in ipairs(espConnections[plr]) do
            conn:Disconnect()
        end
    end
    
    espConnections[plr] = {}
    
    -- Cria ESP se o personagem já existe
    if plr.Character then
        createESP(plr)
    end
    
    -- Conexão para quando o personagem for adicionado
    local charAddedConn
    charAddedConn = plr.CharacterAdded:Connect(function(char)
        -- Pequena pausa para o personagem carregar
        task.wait(0.5)
        createESP(plr)
    end)
    table.insert(espConnections[plr], charAddedConn)
    
    -- Conexão para quando o personagem for removido
    local charRemovingConn
    charRemovingConn = plr.CharacterRemoving:Connect(function()
        removeESP(plr)
    end)
    table.insert(espConnections[plr], charRemovingConn)
end

-- Função principal para habilitar ESP
local function enableESP()
    -- Limpa conexões antigas
    for plr, conns in pairs(espConnections) do
        for _, conn in ipairs(conns) do
            conn:Disconnect()
        end
    end
    table.clear(espConnections)
    
    -- Configura ESP para todos os jogadores existentes
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= player then
            setupPlayerESP(plr)
        end
    end
    
    -- Conexão para novos jogadores que entrarem
    local playerAddedConn
    playerAddedConn = Players.PlayerAdded:Connect(function(plr)
        if plr ~= player then
            setupPlayerESP(plr)
        end
    end)
    
    -- Armazena a conexão principal
    if not espConnections["Main"] then
        espConnections["Main"] = {}
    end
    table.insert(espConnections["Main"], playerAddedConn)
end

-- Inicia o ESP
enableESP()

-- Reconecta se o jogador local respawnar (opcional)
player.CharacterAdded:Connect(function()
    task.wait(1)
    -- Recria ESP para todos quando o player local respawnar
    enableESP()
end)

local function updateBaseESP()
	if not plotsFolder then return end

	for _, plot in ipairs(plotsFolder:GetChildren()) do
		local purchases = plot:FindFirstChild("Purchases")
		local plotBlock = purchases and purchases:FindFirstChild("PlotBlock")
		local mainPart = plotBlock and plotBlock:FindFirstChild("Main")
		local billboard = baseEspInstances[plot.Name]

		local timeLabel = mainPart
			and mainPart:FindFirstChild("BillboardGui")
			and mainPart.BillboardGui:FindFirstChild("RemainingTime")

		if timeLabel and mainPart then
			billboard = billboard or createBaseESP(plot, mainPart)
			local label = billboard:FindFirstChildWhichIsA("TextLabel")
			if label then
				label.Text = timeLabel.Text
			end
		elseif billboard then
			billboard:Destroy()
			baseEspInstances[plot.Name] = nil
		end
	end
end

-- Função para achar plot mais próximo
local function acharPlotMaisProximo(coord)
    local plots = workspace:WaitForChild("Plots")
    local maisProximo = nil
    local menorDistancia = math.huge

    for _, plot in pairs(plots:GetChildren()) do
        if plot:IsA("Model") then
            local primary = plot:FindFirstChildWhichIsA("BasePart", true)
            if primary then
                local distancia = (primary.Position - coord).Magnitude
                if distancia < menorDistancia then
                    menorDistancia = distancia
                    maisProximo = plot
                end
            end
        end
    end

    return maisProximo, menorDistancia
end

-- Função para ativar a base
local function ativarBase(coord)
    local plot, distancia = acharPlotMaisProximo(coord)
    
    if plot and distancia <= DISTANCIA_MAXIMA then
        local prompt = plot:FindFirstChild("UnlockBase", true)
        if prompt and prompt:IsA("ProximityPrompt") then
            fireproximityprompt(prompt)
            else
            warn("Prompt não encontrado")
        end
    else
        warn("Nenhum plot dentro da distância permitida")
    end
end
